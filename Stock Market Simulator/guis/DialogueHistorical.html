130     private void button1ActionPerformed(java.awt.event.ActionEvent evt) throws PropertyVetoException {                                        
131         StockAnalysisTool prediction = new StockAnalysisTool(); // prediciton object initialised for simulation - see StockAnalysisTool Class
132         prediction.symbol = textField1.getText(); //company name entered by user
133         prediction.startDate = jFormattedTextField1.getText(); // start date of historical time series used in prediction entered by the user
134         prediction.endDate = jFormattedTextField2.getText(); // end date of historical time series used in prediction entered by the user
135         dispose(); // dispose dialog
136         
137         
138         try {
139             LineChart graph = prediction.getHistorical(); //generate graph of historical values
140             graph.setSelected(true);
141             Main.menu.add(graph); // add graph to main interface
142             Main.menu.pack();
143             
144             dispose();
145         } catch (ParseException ex) {
146             Logger.getLogger(DialogueHistorical.class.getName()).log(Level.SEVERE, null, ex);
147         }
148          
149     }
