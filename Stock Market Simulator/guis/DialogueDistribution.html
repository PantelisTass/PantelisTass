184     public void button1ActionPerformed(java.awt.event.ActionEvent evt) throws PropertyVetoException {                                        
185          StockAnalysisTool prediction = new StockAnalysisTool(); // prediciton object initialised for simulation - see StockAnalysisTool Class
186         prediction.symbol = textField1.getText(); //company name entered by user
187         prediction.startDate = jFormattedTextField1.getText(); // start date of historical time series used in prediction entered by the user
188         prediction.endDate = jFormattedTextField2.getText(); // end date of historical time series used in prediction entered by the user
189         prediction.val = Integer.parseInt(textField2.getText()); //number of days for which future prices will be generated entered by the user
190         prediction.N = Integer.parseInt(textField5.getText()); // number of projections entered by the user
191         int i = Integer.parseInt(textField3.getText()); // number of intervals ('bins') which are used to classify stock prices
192         dispose(); // dispose dialog
193         try {
194             
195             LineChart graph = prediction.getDistribution(i); //produce graph
196             graph.setSelected(true);
197             Main.menu.add(graph); // add graph to main menu interface
198             Main.menu.pack();
199         } catch (ParseException ex) {
200             Logger.getLogger(DialogueDistribution.class.getName()).log(Level.SEVERE, null, ex);
201         }
202         dispose();
203     }
