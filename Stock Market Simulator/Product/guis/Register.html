131     private void button1ActionPerformed(java.awt.event.ActionEvent evt) {// method handles registration                               
132        
133         String pass = new String(jPasswordField1.getPassword()); //password
134         String user = new String(textField1.getText()); // username
135         if(ManagePortfolio.UserCheck(user,pass)==false){ // if no match made with database
136             ManagePortfolio.createUser(user, pass); //create no user
137             reset();
138             dispose();
139             menu = new Menu(user); // redirect to ain interface - see Menu Class
140             menu.setTitle("Dashboard");
141             menu.setVisible(true);
142         
143         }
144         else if(ManagePortfolio.UserCheck(user,pass)==true){ // if match made wih database
145                 JOptionPane.showMessageDialog(null,
146                         "Username already taken!",
147                         "Error",
148                         JOptionPane.ERROR_MESSAGE);
149                         dispose(); //prevent access --> display error dialogue
