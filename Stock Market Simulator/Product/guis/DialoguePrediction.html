163     private void button1ActionPerformed(java.awt.event.ActionEvent evt) throws PropertyVetoException {                                        
164         StockAnalysisTool prediction = new StockAnalysisTool(); // prediciton object initialised for simulation - see StockAnalysisTool Class
165         prediction.symbol = textField1.getText(); //company name entered by user
166         prediction.startDate = jFormattedTextField1.getText(); // start date of historical time series used in prediction entered by the user
167         prediction.endDate = jFormattedTextField2.getText(); // end date of historical time series used in prediction entered by the user
168         prediction.val = Integer.parseInt(textField2.getText()); //number of days for which future prices will be generated entered by the user
169         prediction.N = Integer.parseInt(textField5.getText()); // number of projections entered by the user
170         dispose(); // dispose dialog
171         try {
172             LineChart graph = prediction.DisplayFuture();
173             graph.setSelected(true);
174             Main.menu.add(graph); //add graph of prediction to main JFrame
175             Main.menu.pack();
176         } catch (ParseException ex) {
177             Logger.getLogger(DialogueHistorical.class.getName()).log(Level.SEVERE, null, ex);
178         }
179 
180     }
