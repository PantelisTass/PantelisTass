28     public static byte[] getSHA(String password) throws NoSuchAlgorithmException 
29     {  
30         
31         // Static getInstance method is called with hashing SHA  
32         MessageDigest md = MessageDigest.getInstance("SHA-256");  
33   
34         // digest() method called  
35         // to calculate message digest (A digest is a small value generated by a hash function from a whole message) of an input  
36         // and return array of byte 
37         return md.digest(password.getBytes(StandardCharsets.UTF_8));  
38     }

40     public static String toHexString(byte[] hash) 
41     { 
42         // Convert byte array into signum representation  
43         BigInteger number = new BigInteger(1, hash);  
44   
45         // Convert message digest into hex value  
46         StringBuilder hexString = new StringBuilder(number.toString(16));  
47   
48         // Pad with leading zeros 
49         while (hexString.length() < 32)  
50         {  
51             hexString.insert(0, '0');  
52         }  
53   
54         return hexString.toString();  
55     }
